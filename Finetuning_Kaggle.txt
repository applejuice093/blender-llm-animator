   "cell_type": "code",
   "execution_count": null,

   "source": [
    "import os\n",
    "import json\n",
    "import glob\n",
    "import torch\n",
    "from unsloth import FastLanguageModel\n",
    "from trl import SFTTrainer\n",
    "from transformers import TrainingArguments\n",
    "from datasets import Dataset\n",
    "\n",
    "MAX_SEQ_LENGTH = 2048\n",
    "MODEL_NAME = \"unsloth/Qwen2.5-3B-Instruct\"\n",
    "OUTPUT_DIR = \"/kaggle/working/lora_output\"\n",
    "\n",
    "def load_data():\n",
    "    records = []\n",
    "    # Search all jsonl files in current dir, /kaggle/working, or /kaggle/input\n",
    "    search_paths = glob.glob(\"*.jsonl\") + glob.glob(\"datasets/*.jsonl\") + glob.glob(\"/kaggle/working/*.jsonl\") + glob.glob(\"/kaggle/input/**/*.jsonl\", recursive=True)\n",
    "    search_paths = list(set(search_paths))\n",
    "    \n",
    "    print(f\"Scanning dataset paths: {search_paths}\")\n",
    "    for filepath in search_paths:\n",
    "        if not os.path.isfile(filepath):\n",
    "            continue\n",
    "        with open(filepath, \"r\") as f:\n",
    "            for line in f:\n",
    "                if not line.strip():\n",
    "                    continue\n",
    "                try:\n",
    "                    record = json.loads(line)\n",
    "                    # Pick fixed_plan if self-repair succeeded, otherwise polished_plan\n",
    "                    final_plan = record.get(\"fixed_plan\") if record.get(\"fixed_validation_success\") else record.get(\"polished_plan\")\n",
    "                    if not final_plan:\n",
    "                        final_plan = record.get(\"plan\")\n",
    "                        \n",
    "                    is_valid = record.get(\"validation_success\") or record.get(\"fixed_validation_success\") or (\"validation_success\" not in record)\n",
    "                    \n",
    "                    if final_plan and is_valid:\n",
    "                        instruction = f\"Character: {record.get('character', 'Gwen')}\\nRequest: {record['prompt']}\\nGenerate the JSON animation plan.\"\n",
    "                        response = json.dumps(final_plan)\n",
    "                        records.append({\n",
    "                            \"instruction\": instruction,\n",
    "                            \"output\": response\n",
    "                        })\n",
    "                except Exception as e:\n",
    "                    continue\n",
    "    if len(records) > 0:\n",
    "        return Dataset.from_list(records)\n",
    "    return None\n",
    "\n",
    "def format_prompt(examples):\n",
    "    system_prompt = \"You are an expert animation director. You must output a valid JSON object representing an animation intent plan, and NOTHING ELSE.\"\n",
    "    texts = []\n",
    "    for instruction, output in zip(examples[\"instruction\"], examples[\"output\"]):\n",
    "        text = f\"<|im_start|>system\\n{system_prompt}<|im_end|>\\n<|im_start|>user\\n{instruction}<|im_end|>\\n<|im_start|>assistant\\n{output}<|im_end|>\"\n",
    "        texts.append(text)\n",
    "    return {\"text\": texts}\n",
    "\n",
    "print(\"Loading dataset records...\")\n",
    "dataset = load_data()\n",
    "if not dataset:\n",
    "    raise ValueError(\"No valid training records found! Make sure dataset .jsonl files are available in Kaggle.\")\n",
    "\n",
    "print(f\"Found {len(dataset)} valid training examples. Formatting...\")\n",
    "dataset = dataset.map(format_prompt, batched=True)\n",
    "\n",
    "print(f\"Loading Base Model {MODEL_NAME} in 4-bit...\")\n",
    "model, tokenizer = FastLanguageModel.from_pretrained(\n",
    "    model_name = MODEL_NAME,\n",
    "    max_seq_length = MAX_SEQ_LENGTH,\n",
    "    dtype = None,\n",
    "    load_in_4bit = True,\n",
    ")\n",
    "\n",
    "model = FastLanguageModel.get_peft_model(\n",
    "    model,\n",
    "    r = 16,\n",
    "    target_modules = [\"q_proj\", \"k_proj\", \"v_proj\", \"o_proj\",\n",
    "                      \"gate_proj\", \"up_proj\", \"down_proj\"],\n",
    "    lora_alpha = 16,\n",
    "    lora_dropout = 0,\n",
    "    bias = \"none\",\n",
    "    use_gradient_checkpointing = \"unsloth\",\n",
    "    random_state = 3407,\n",
    "    use_rslora = False,\n",
    "    loftq_config = None,\n",
    ")\n",
    "\n",
    "# Adjust max_steps based on dataset size (approx 2 epochs)\n",
    "calculated_steps = max(60, (len(dataset) // 8) * 2)\n",
    "print(f\"Training for {calculated_steps} steps...\")\n",
    "\n",
    "trainer = SFTTrainer(\n",
    "    model = model,\n",
    "    tokenizer = tokenizer,\n",
    "    train_dataset = dataset,\n",
    "    dataset_text_field = \"text\",\n",
    "    max_seq_length = MAX_SEQ_LENGTH,\n",
    "    dataset_num_proc = 2,\n",
    "    packing = False,\n",
    "    args = TrainingArguments(\n",
    "        per_device_train_batch_size = 2,\n",
    "        gradient_accumulation_steps = 4,\n",
    "        warmup_steps = 5,\n",
    "        max_steps = calculated_steps,\n",
    "        learning_rate = 2e-4,\n",
    "        fp16 = not torch.cuda.is_bf16_supported(),\n",
    "        bf16 = torch.cuda.is_bf16_supported(),\n",
    "        logging_steps = 1,\n",
    "        optim = \"adamw_8bit\",\n",
    "        weight_decay = 0.01,\n",
    "        lr_scheduler_type = \"linear\",\n",
    "        seed = 3407,\n",
    "        output_dir = \"outputs\",\n",
    "        average_tokens_across_devices = False,\n",
    "    ),\n",
    ")\n",
    "\n",
    "print(\"Starting training...\")\n",
    "trainer.train()\n",
    "print(\"Training complete!\")\n",
    "\n",
    "print(\"Saving trained adapters...\")\n",
    "model.save_pretrained(OUTPUT_DIR)\n",
    "tokenizer.save_pretrained(OUTPUT_DIR)\n",
    "\n",
    "print(\"Exporting to GGUF (q4_k_m)...\")\n",
    "try:\n",
    "    model.save_pretrained_gguf(OUTPUT_DIR, tokenizer, quantization_method = \"q4_k_m\")\n",
    "    print(f\"SUCCESS! Exported GGUF file saved inside {OUTPUT_DIR}\")\n",
    "except Exception as e:\n",
    "    print(f\"GGUF Export note: {e}\")"
   ]
  }
