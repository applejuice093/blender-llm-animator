    "  \"camera\": {\n",

    "    \"movement\": \"string\"\n",

    "  \"notes\": \"string\"\n",
    "}\n",
    "\"\"\"\n",
    "\n",
    "def compile_script(plan: dict) -> str:\n",
    "    # Generates standard python code for the mock scene\n",
    "    script = \"import bpy\\nimport math\\n\\n\"\n",
    "    script += f\"# Compiled Animation Plan for {plan.get('character')}\\n\\n\"\n",
    "    \n",
    "    emotion = plan.get(\"emotion\", {})\n",
    "    script += f\"# Set Emotion: {emotion.get('primary')} (intensity {emotion.get('intensity')})\\n\"\n",
    "    script += f\"armature = bpy.data.objects.get('{RIG_NAME}')\\n\"\n",
    "    script += f\"if armature:\\n    pass\\n\\n\"\n",
    "    \n",
    "    for g in plan.get(\"gestures\", []):\n",
    "        start_frame = int(g.get(\"start\", 0.0) * 24)\n",
    "        end_frame = int(g.get(\"end\", 1.0) * 24)\n",
    "        script += f\"# Gesture: {g.get('type')} from {g.get('start')}s (f{start_frame}) to {g.get('end')}s (f{end_frame})\\n\"\n",
    "        script += f\"if armature:\\n    pass\\n\\n\"\n",
    "        \n",
    "    camera = plan.get(\"camera\", {})\n",
    "    script += f\"# Camera: {camera.get('shot')}\\n\"\n",
    "    script += f\"cam = bpy.data.objects.get('Camera')\\n\"\n",
    "    script += f\"if cam:\\n    pass\\n\\n\"\n",
    "    return script\n",
    "\n",
    "def validate_script_headlessly(script_code: str) -> tuple[bool, str]:\n",
    "    safe_script = script_code + \"\\n\\nimport sys\\nsys.exit(0)\\n\"\n",
    "    with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:\n",
    "        f.write(safe_script)\n",
    "        temp_path = f.name\n",
    "    try:\n",
    "        result = subprocess.run([BLENDER_PATH, \"-b\", \"-P\", temp_path], capture_output=True, text=True, timeout=15)\n",
    "        os.remove(temp_path)\n",
    "        if result.returncode == 0:\n",
    "            return True, \"\"\n",
    "        return False, result.stderr if result.stderr else result.stdout\n",
    "    except Exception as e:\n",
    "        if os.path.exists(temp_path):\n",
    "            os.remove(temp_path)\n",
    "        return False, str(e)\n",
    "\n",
    "# =====================================================================\n",
    "# 4. LOAD MODEL (QWEN-7B VIA UNSLOTH)\n",

    "print(\"Loading Qwen2.5-7B-Instruct in 4-bit...\")\n",
    "model, tokenizer = FastLanguageModel.from_pretrained(\n",

    "def call_model(prompt: str, system: str) -> dict:\n",
    "    messages = [\n",
    "        {\"role\": \"system\", \"content\": system},\n",
    "        {\"role\": \"user\", \"content\": prompt}\n",
    "    ]\n",
    "    inputs = tokenizer.apply_chat_template(messages, tokenize=True, add_generation_prompt=True, return_tensors=\"pt\").to(\"cuda\")\n",
    "    with torch.no_grad():\n",
    "        outputs = model.generate(inputs, max_new_tokens=1024, use_cache=True)\n",
    "    resp = tokenizer.decode(outputs[0][len(inputs[0]):], skip_special_tokens=True).strip()\n",
    "    \n",
    "    # Clean markdown code blocks\n",
    "    if resp.startswith(\"```\"):\n",
    "        lines = resp.split(\"\\n\")\n",
    "        if lines[0].startswith(\"```\"):\n",
    "            lines = lines[1:]\n",
    "        if lines and lines[-1].strip() == \"```\":\n",
    "            lines = lines[:-1]\n",
    "        resp = \"\\n\".join(lines).strip()\n",
    "        \n",
    "    try:\n",
    "        return json.loads(resp)\n",
    "    except:\n",
    "        match = re.search(r'\\{.*\\}', resp, re.DOTALL)\n",
    "        if match:\n",
    "            try:\n",
    "                return json.loads(match.group(0))\n",
    "            except:\n",
    "                pass\n",
    "        print(f\"Raw generation failed to parse as JSON: {resp[:200]}...\")\n",
    "        return {}\n",

    "    fixed_plan = None\n",

    "    fixed_validation_error = None\n",

    "        if fixed_validation_success:\n",

    "            script = fixed_script\n",
    "        else:\n",
    "            print(\"[-] Self-Repair Failed.\")\n",
    "            \n",
    "    # Write Record to JSONL\n",
    "    record = {\n",
